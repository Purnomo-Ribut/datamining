dataset = dataset.drop(columns=["Unnamed: 0"])
dataset = dataset.replace(",",".",regex=True)

from sklearn.model_selection import train_test_split

X = dataset.iloc[:, :-1].astype(float)

y = dataset.Approved.astype(int)

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size =0.2, random_state=0)

#NAIVE BAIYES
from sklearn.naive_bayes import GaussianNB
from sklearn.metrics import accuracy_score

#modeling
model=GaussianNB()
model.fit(X_train, y_train)
y_pred=model.predict(X_test)
accuracy =accuracy_score(y_test,y_pred)
print ("Acuracy Naive Baiyes = ", accuracy)

Acuracy Naive Baiyes =  1.0

#KNN
from sklearn.neighbors import KNeighborsClassifier
# Create KNN classifier
knn = KNeighborsClassifier(n_neighbors = 3)
# Fit the classifier to the data
knn.fit(X_train,y_train)
y_pred=knn.predict(X_test)
accuracy =accuracy_score(y_test,y_pred)
print ("Acuracy KNN = ", accuracy)

Acuracy KNN =  0.9333333333333333

#DECISSION TREE
from sklearn.tree import DecisionTreeClassifier

### Predicting the Test set results
y_pred = decision_tree_classifier.predict(X_test)

print ("Acuracy Decision Tree = ", accuracy)
